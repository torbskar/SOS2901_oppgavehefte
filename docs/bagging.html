::: {.solution}


```{r}
fmla <- default ~ age + amount + percent_of_income + purpose + employment_duration + housing 

fmla

set.seed(1)
bag <- bagging(
  formula = fmla,
  data = credit,
  nbagg = 150,   
  coob = TRUE,
  control = rpart.control(minsplit = 2, cp = 0)
)
bag
```
